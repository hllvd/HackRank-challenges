The player with the highest score is ranked number 1 on the leaderboard. Players who have equal scores receive the same ranking number, and the next player(s) receive the immediately following ranking number.
ranked = [100,90,90,80]
payer = [70,80,105] The ranked players will have ranks 1,2,2 and respectively. If the player's scores are 70, 80 and 105, their rankings after each game are 4, 3 and 1. Return [4,3,1]. 
scores = ("28 25 22 15 14 12 6 4").split(" ").map(r=>Number(r))
alice = ("5 5 6 14 19 20 23 25 29 29").split(" ").map(r=>Number(r))

scores = [...new Set(scores)]
var jump = 0
var positions = []
for( let a of alice){
    for( let s=scores.length-jump-1;s>-1;s--){
        if(a < scores[s] ){
            break;
        }
        if(a >= scores[s]|| s == 0 ){
            jump = scores.length - s
        }
    }
    positions.push( (scores.length-jump+1))
}
